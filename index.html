    <script>
        // --- Sakura Particles ---
        const canvas = document.getElementById('sakura-canvas');
        const ctx = canvas.getContext('2d');
        let petals = [];
        let celebrating = false;

        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        window.addEventListener('resize', resizeCanvas);
        resizeCanvas();

        class Petal {
            constructor() { this.reset(); }
            reset() {
                this.x = Math.random() * canvas.width;
                this.y = celebrating ? canvas.height + 20 : -20;
                this.w = 8 + Math.random() * 12;
                this.h = this.w * 0.8;
                this.opacity = 0.5 + Math.random() * 0.5;
                this.speed = (celebrating ? -5 : 1) - Math.random() * 2;
                this.curve = Math.random() * 2;
                this.angle = Math.random() * Math.PI * 2;
                this.phi = Math.random() * Math.PI * 2;
            }
            update() {
                this.y += this.speed;
                this.x += Math.sin(this.phi) * this.curve;
                this.phi += 0.01;
                this.angle += 0.02;
                if (!celebrating && this.y > canvas.height) this.reset();
                if (celebrating && this.y < -50) this.reset();
            }
            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate(this.angle);
                ctx.beginPath();
                ctx.ellipse(0, 0, this.w / 2, this.h / 2, 0, 0, Math.PI * 2);
                ctx.fillStyle = `rgba(255, 183, 197, ${this.opacity})`;
                ctx.fill();
                ctx.restore();
            }
        }

        for (let i = 0; i < 40; i++) petals.push(new Petal());

        function animateSakura() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            petals.forEach(p => { p.update(); p.draw(); });
            requestAnimationFrame(animateSakura);
        }
        animateSakura();

        function startCelebration() {
            celebrating = true;
            setTimeout(() => { celebrating = false; }, 3000);
        }

        // --- Logic ---
        let selectedSticker = '🌸';
        let works = [];
        let readCount = 0;

            if (saved) {
                works = JSON.parse(saved);
                renderGallery();
            }
        };

            const haruMsg = document.getElementById('haru-msg');
            const messages = [

        function showTip(term, tip) {
            showModal(term + "の豆知識", tip, "💡");
            readCount++;
            if (readCount >= 3 && !badges.read) {
                badges.read = true;
                document.getElementById('badge-1').classList.remove('grayscale');
                document.getElementById('badge-1').classList.add('badge-get');
            }
        }

        function selectSticker(emoji) {
            selectedSticker = emoji;
            document.querySelectorAll('.sticker-item').forEach(el => el.classList.remove('sticker-selected'));
            event.currentTarget.classList.add('sticker-selected');
            document.getElementById('preview-sticker').innerText = emoji;
        }

        function switchTab(tab) {
            document.getElementById('sec-learn').classList.add('hidden');
            document.getElementById('sec-write').classList.add('hidden');
            document.getElementById('sec-gallery').classList.add('hidden');
            ['learn', 'write', 'gallery'].forEach(t => {
                document.getElementById(`tab-${t}`).className = 'px-3 md:px-5 py-2 rounded-lg font-bold transition-all text-gray-500 hover:bg-white/60';
            });
            document.getElementById(`sec-${tab}`).classList.remove('hidden');
            document.getElementById(`tab-${tab}`).className = 'px-3 md:px-5 py-2 rounded-lg font-bold transition-all tab-active';
        }

        function saveWork() {
            const author = document.getElementById('author').value.trim();
            const content = document.getElementById('work-content').value.trim();
            const comment = document.getElementById('work-comment').value.trim();
            const type = document.querySelector('input[name="type"]:checked').value;

            if (!author || !content) {
                showModal("お名前を教えて！", "作品となまえを書いてからボタンをおしてね。", "📝");
                return;
            }

            const newWork = {
                id: Date.now(),
                author, content, comment, type, sticker: selectedSticker,
                date: new Date().toLocaleDateString('ja-JP'),
                feedback: []
            };

            works.unshift(newWork);
            saveData();
            
            renderGallery();
            startCelebration();
            
            if (!badges.posted) {
                badges.posted = true;
                document.getElementById('badge-2').classList.remove('grayscale');
                document.getElementById('badge-2').classList.add('badge-get');
            }

            showModal("やったね！", "作品がギャラリーにならんだよ！", "🎉");
            document.getElementById('work-content').value = '';
            document.getElementById('work-comment').value = '';
            setTimeout(() => { closeModal(); switchTab('gallery'); }, 1500);
        }

        function addFeedback(workId) {
            const input = document.getElementById(`feedback-input-${workId}`);
            const text = input.value.trim();
            if (!text) return;

            const work = works.find(w => w.id === workId);
            if (work) {
                work.feedback.push(text);
                saveData();
                renderGallery();
                talkHaruChan("すてきな感想をありがとう！");
                
                if (!badges.commented) {
                    badges.commented = true;
                    document.getElementById('badge-3').classList.remove('grayscale');
                    document.getElementById('badge-3').classList.add('badge-get');
                }
            }
        }

        function saveData() {
            localStorage.setItem('spring_works_v3', JSON.stringify(works));
        }

        function renderGallery() {
            const container = document.getElementById('gallery-container');
            const emptyMsg = document.getElementById('empty-msg');
            container.querySelectorAll('.work-card').forEach(c => c.remove());

            if (works.length === 0) {
                emptyMsg.style.display = 'block';
                return;
            }

            emptyMsg.style.display = 'none';

            works.forEach(work => {
                const card = document.createElement('div');
                card.className = 'work-card glass p-6 rounded-[40px] shadow-lg border-none flex flex-col items-center bounce-hover relative overflow-hidden';
                
                let feedbackHtml = work.feedback.map(f => `<div class="comment-bubble mincho">${f}</div>`).join('');
                
                card.innerHTML = `
                    <div class="absolute -top-2 -left-2 w-16 h-16 bg-pink-100 rounded-full flex items-center justify-center text-3xl opacity-40 select-none">
                        ${work.sticker}
                    </div>
                    <div class="w-full flex justify-between items-start mb-4">
                        <span class="text-[10px] font-bold px-3 py-1 bg-pink-500 text-white rounded-full">${work.type}</span>
                    </div>
                    <div class="h-56 flex items-center justify-center mb-6 border-r-2 border-dashed border-pink-100 pr-6">
                        <div class="vertical-text mincho text-2xl text-gray-700 leading-loose h-full tracking-widest">
                            ${work.content.replace(/\n/g, '<br>')}
                        </div>
                    </div>
                    <div class="w-full pt-4 border-t border-pink-50 text-center">
                        <p class="font-bold text-gray-800">${work.author} さんの作品</p>
                    </div>

                    <!-- Feedback Area -->
                    <div class="w-full mt-6 space-y-4">
                        <div class="max-h-32 overflow-y-auto px-2">
                            ${feedbackHtml}
                        </div>
                        <div class="flex gap-2">
                            <input type="text" id="feedback-input-${work.id}" placeholder="感想をおくろう！" 
                                class="flex-1 text-xs p-2 rounded-xl border-none ring-1 ring-pink-100 focus:ring-2 focus:ring-pink-300 outline-none bg-white/80">
                            <button onclick="addFeedback(${work.id})" class="bg-pink-400 text-white p-2 rounded-xl hover:bg-pink-500 transition-colors">
                                📮
                            </button>
                        </div>
                    </div>
                `;
                container.appendChild(card);
            });
        }

        function clearGallery() {
            if (confirm('リセットしてもいいですか？')) {
                works = [];
                localStorage.removeItem('spring_works_v3');
                renderGallery();
            }
        }

        function showModal(title, text, icon) {
            const modal = document.getElementById('modal');
            const content = document.getElementById('modal-content');
            document.getElementById('modal-title').innerText = title;
            document.getElementById('modal-text').innerText = text;
            document.getElementById('modal-icon').innerText = icon;
            modal.classList.remove('hidden');
            setTimeout(() => { content.classList.remove('scale-95', 'opacity-0'); content.classList.add('scale-100', 'opacity-100'); }, 10);
        }

        function closeModal() {
            const modal = document.getElementById('modal');
            const content = document.getElementById('modal-content');
            content.classList.remove('scale-100', 'opacity-100');
            content.classList.add('scale-95', 'opacity-0');
            setTimeout(() => { modal.classList.add('hidden'); }, 300);
        }
